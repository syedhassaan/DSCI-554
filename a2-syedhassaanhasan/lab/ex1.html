🐞 ex1: debugging Javascript

0 - How to

TL;DR. Escalate as needed
1. console.log() & hot-reload → 2. Chrome debugger → 3. VS Code & Chrome debugger
  1. Add console.log() in your code and inspect the output in the Chrome Dev Tools Console: 
    
var a = "Hello world"
console.log(a)
  2. Chrome debugger: set breakpoints in the code in Chrome Dev Tools and reload the page
    1. Chrome Dev Tools → Sources tab → open the file to debug
    2. Set breakpoints
    3. Reload the page
    4. Mouse over variables and or use the Console prompt to inspect
    ⚠️ console.log() is not available at the Console prompt, simply type the variable name.
    💡 See also debugging with Chrome
  3. VS Code & Chrome debugger configured through launch.json:
    1. Install Debugger for Chrome extension
    2. Set breakpoints
    3. Click Run button on the side bar
    4. Select launcher to run using the dropdown
    5. Click ▷
    4. 🚦 launch.json is ready to go for debugging the lab exercises!
    💡 Learn more about debugging with VS Code here

💡 Javascript object are defined as comma separated list of key-value pairs: {🔑: 🇺🇸, 🗝: 🇪🇸}

💡 Javascript arrays are defined as a comma separated list of values: [🍎, 🍌, 🍉, 🍊, 🍍, 🥭]

💡 Javascript functions are defined the old way as:
function(food) { console.log('I like ' + food); }

💡 Javascript functions are defined with 🏹 notation as:
(food) => { console.log('I like ' + food); }
Sample code 

var v;  //variable declaration, v = undefined
v = 'a';  //variable initialization
var v = 'a';  //declaration with initialization

// Dynamic typing of primitive types
var foo = true;       //Boolean
var foo = null;       //Null
var foo = undefined;  //Undefined
var foo = 2.3;        //Number
var foo = 'bar'       //String, specified using '' or ""

var arr = ['tree', 1765, [0, 1, 2]];  //array
var a0 = arr[0];  //access first element in array (indices starts at 0)
console.log(JSON.stringify(arr));  //prints nested array

var obj = { first: 'Joseph', last: 'Priestley' };  //object literal

//console.log(obj.first);  //access property first of obj (preferred!)
//console.log(obj['first']);  //access property first of obj (only use if the property includes spaces!)

var name = obj.first + ' ' + obj.last;  //+ concatenates Strings
var x = 3;
console.log('x', '=', x);  //, concatenates Strings adding a space

for (i = 0; i < arr.length; i++) {  //for loop to iterate over an array
  console.log(arr[i]);
}

arr.forEach(function (d) {  //forEach is cleaner to use to iterate over an array!
  console.log(d);
});

arr.forEach(d => {  //same with arrow function notation
  console.log(d);
});

if (true) {  //conditional statement
  console.log('right is right');
} else {
  console.log('wrong is wrong');
}

//same with ternary operator
console.log(true ? 'right is right' : 'wrong is wrong'); 

function add1(a, b) {   //function declaration
  return a + b; 
}  //no ; at the end!

var add2 = function (a, b) {  //function expression
  return a + b;
};  //; at the end!

var sum1 = add1(1, 2);  //function call

console.log('==============================================');
