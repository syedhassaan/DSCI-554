Starter code 

<!-- Update this preview -->		
<style>
div.bar {
  background-color: orangered;
  margin-bottom: 6px;

<!-- load d3 library (referenced in package.json)! -->
<script src="../node_modules/d3/dist/d3.min.js"></script>

<div id="chart"></div>

<script>
var el = d3.select('#chart');
el.append('div')
  .text('100 pixels div')
  .attr('class', 'bar')  //class is an attribute, so we use .attr()
  .style('width', '100px');
</script>
